```bash
# scp -l 代表 limit Kbit/s，即 1024*8*xxx xxx mb/s，限速 scp 拷贝

# （通过密码） scp 从服务器复制文件夹到本地路径样例
scp -r -P 20022 xxx@xx.xx.xx.xx:/source_path target_path

# （通过私钥） scp 从服务器复制文件夹到本地路径样例
scp -r -P 20022 -i "C:\xx\xx" xxx@xx.xx.xx.xx:/source_path target_path


# （通过私钥） scp 从本地路径复制文件夹到服务器样例
scp -r -i "C:\xx\xx" "source_path" xxx@xx.xx.xx.xx:/target_path
```
